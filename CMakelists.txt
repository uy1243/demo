# Windows 自动复制 DLL 到运行目录

    set(OUT_DIR "$<TARGET_FILE_DIR:trader>")
    
    # 复制旧 API 的 DLL
    add_custom_command(TARGET trader POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different "${MYSQL_BIN_DIR}/${MYSQL_OLD_DLL}" ${OUT_DIR}

    set(TEST_OUT_DIR "$<TARGET_FILE_DIR:backtest>")
    
    add_custom_command(TARGET backtest POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different "${MYSQL_BIN_DIR}/${MYSQL_OLD_DLL}" ${TEST_OUT_DIR}
