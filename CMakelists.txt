cmake_minimum_required(VERSION 3.10)
project(futures_ctp)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fPIC -Wall -Wno-unused-variable")

include_directories(${PROJECT_SOURCE_DIR}/api)
link_directories(${PROJECT_SOURCE_DIR}/api/lib)

add_executable(futures_trader
    src/main.cpp
    src/TraderSpi.cpp
    src/MdSpi.cpp
    src/Logger.cpp
)

target_link_libraries(futures_trader
    thosttraderapi
    thostmduserapi
    pthread
    dl
)
